54  /**
55   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
56   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
57   * to the <strong>"SecureChat"</strong> example in the distribution or the web
58   * site for the detailed usage.
59   *
60   * <h3>Beginning the handshake</h3>
61   * <p>
62   * You must make sure not to write a message while the
63   * {@linkplain #handshake(Channel) handshake} is in progress unless you are
64   * renegotiating.  You will be notified by the {@link ChannelFuture} which is
65   * returned by the {@link #handshake(Channel)} method when the handshake
66   * process succeeds or fails.
67   *
68   * <h3>Renegotiation</h3>
69   * <p>
70   * Once the initial handshake is done successfully.  You can always call
71   * {@link #handshake(Channel)} again to renegotiate the SSL session parameters.
72   *
73   * <h3>Closing the session</h3>
74   * <p>
75   * To close the SSL session, the {@link #close(Channel)} method should be
76   * called to send the {@code close_notify} message to the remote peer.  One
77   * exception is when you close the {@link Channel} - {@link SslHandler}
78   * intercepts the close request and send the {@code close_notify} message
79   * before the channel closure automatically.  Once the SSL session is closed,
80   * it is not reusable, and consequently you should create a new
81   * {@link SslHandler} with a new {@link SSLEngine} as explained in the
82   * following section.
83   *
84   * <h3>Restarting the session</h3>
85   * <p>
86   * To restart the SSL session, you must remove the existing closed
87   * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
88   * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
89   * and start the handshake process as described in the first section.
90   *
91   * <h3>Implementing StartTLS</h3>
92   * <p>
93   * <a href="http://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
94   * communication pattern that secures the wire in the middle of the plaintext
95   * connection.  Please note that it is different from SSL &middot; TLS, that
96   * secures the wire from the beginning of the connection.  Typically, StartTLS
97   * is composed of three steps:
98   * <ol>
99   * <li>Client sends a StartTLS request to server.</li>
100  * <li>Server sends a StartTLS response to client.</li>
101  * <li>Client begins SSL handshake.</li>
102  * </ol>
103  * If you implement a server, you need to:
104  * <ol>
105  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
106  *     to {@code true},</li>
107  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
108  * <li>write a StartTLS response.</li>
109  * </ol>
110  * Please note that you must insert {@link SslHandler} <em>before</em> sending
111  * the StartTLS response.  Otherwise the client can send begin SSL handshake
112  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
113  * data corruption.
114  * <p>
115  * The client-side implementation is much simpler.
116  * <ol>
117  * <li>Write a StartTLS request,</li>
118  * <li>wait for the StartTLS response,</li>
119  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
120  *     to {@code false},</li>
121  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
122  * <li>Initiate SSL handshake by calling {@link SslHandler#handshake(Channel)}.</li>
123  * </ol>
124  *
125  * @author The Netty Project (netty-dev@lists.jboss.org)
126  * @author Trustin Lee (tlee@redhat.com)
127  *
128  * @version $Rev: 1740 $, $Date: 2009-09-22 18:57:26 +0900 (화, 22 9 2009) $
129  *
130  * @apiviz.landmark
131  * @apiviz.uses org.jboss.netty.handler.ssl.SslBufferPool
132  */

302     /**
303      * Starts an SSL / TLS handshake for the specified channel.
304      *
305      * @return a {@link ChannelFuture} which is notified when the handshake
306      *         succeeds or fails.
307      */
308     public ChannelFuture handshake(Channel channel) throws SSLException {
309         ChannelFuture handshakeFuture;
310         synchronized (handshakeLock) {
311             if (handshaking) {
312                 return this.handshakeFuture;
313             } else {
314                 engine.beginHandshake();
315                 runDelegatedTasks();
316                 handshakeFuture = this.handshakeFuture = future(channel);
317                 handshaking = true;
318             }
319         }
320 
321         wrapNonAppData(context(channel), channel);
322         return handshakeFuture;
323     }

339     public void handleDownstream(
340             final ChannelHandlerContext context, final ChannelEvent evt) throws Exception {
341         if (evt instanceof ChannelStateEvent) {
342             ChannelStateEvent e = (ChannelStateEvent) evt;
343             switch (e.getState()) {
344             case OPEN:
345             case CONNECTED:
346             case BOUND:
347                 if (Boolean.FALSE.equals(e.getValue()) || e.getValue() == null) {
348                     closeOutboundAndChannel(context, e);
349                     return;
350                 }
351             }
352         }
353         if (!(evt instanceof MessageEvent)) {
354             context.sendDownstream(evt);
355             return;
356         }
357 
358         MessageEvent e = (MessageEvent) evt;
359         if (!(e.getMessage() instanceof ChannelBuffer)) {
360             context.sendDownstream(evt);
361             return;
362         }
363 
364         // Do not encrypt the first write request if this handler is
365         // created with startTLS flag turned on.
366         if (startTls && sentFirstMessage.compareAndSet(false, true)) {
367             context.sendDownstream(evt);
368             return;
369         }
370 
371         // Otherwise, all messages are encrypted.
372         ChannelBuffer msg = (ChannelBuffer) e.getMessage();
373         PendingWrite pendingWrite;
374 
375         if (msg.readable()) {
376             pendingWrite = new PendingWrite(evt.getFuture(), msg.toByteBuffer(msg.readerIndex(), msg.readableBytes()));
377         } else {
378             pendingWrite = new PendingWrite(evt.getFuture(), null);
379         }
380         synchronized (pendingUnencryptedWrites) {
381             boolean offered = pendingUnencryptedWrites.offer(pendingWrite);
382             assert offered;
383         }
384 
385         wrap(context, evt.getChannel());
386     }

388     @Override
389     public void channelDisconnected(ChannelHandlerContext ctx,
390             ChannelStateEvent e) throws Exception {
391 
392         // Make sure the handshake future is notified when a connection has
393         // been closed during handshake.
394         synchronized (handshakeLock) {
395             if (handshaking) {
396                 handshakeFuture.setFailure(new ClosedChannelException());
397             }
398         }
399 
400         try {
401             super.channelDisconnected(ctx, e);
402         } finally {
403             unwrap(ctx, e.getChannel(), ChannelBuffers.EMPTY_BUFFER, 0, 0);
404             engine.closeOutbound();
405             if (!sentCloseNotify.get() && handshaken) {
406                 try {
407                     engine.closeInbound();
408                 } catch (SSLException ex) {
409                     logger.debug("Failed to clean up SSLEngine.", ex);
410                 }
411             }
412         }
413     }

415     @Override
416     public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e)
417             throws Exception {
418 
419         Throwable cause = e.getCause();
420         if (cause instanceof IOException) {
421             if (cause instanceof ClosedChannelException) {
422                 synchronized (ignoreClosedChannelExceptionLock) {
423                     if (ignoreClosedChannelException > 0) {
424                         ignoreClosedChannelException --;
425                         logger.debug(
426                                 "Swallowing an exception raised while " +
427                                 "writing non-app data", cause);
428                         return;
429                     }
430                 }
431             } else if (engine.isOutboundDone()) {
432                 String message = String.valueOf(cause.getMessage()).toLowerCase();
433                 if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
434                     // It is safe to ignore the 'connection reset by peer' or
435                     // 'broken pipe' error after sending closure_notify.
436                     logger.debug(
437                             "Swallowing a 'connection reset by peer / " +
438                             "broken pipe' error occurred while writing " +
439                             "'closure_notify'", cause);
440 
441                     // Close the connection explicitly just in case the transport
442                     // did not close the connection automatically.
443                     Channels.close(ctx, succeededFuture(e.getChannel()));
444                     return;
445                 }
446             }
447         }
448 
449         ctx.sendUpstream(e);
450     }

452     @Override
453     protected Object decode(
454             final ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer) throws Exception {
455         if (buffer.readableBytes() < 2) {
456             return null;
457         }
458 
459         int packetLength = buffer.getShort(buffer.readerIndex()) & 0xFFFF;
460         if ((packetLength & 0x8000) != 0) {
461             // Detected a SSLv2 packet
462             packetLength &= 0x7FFF;
463             packetLength += 2;
464         } else  if (buffer.readableBytes() < 5) {
465             return null;
466         } else {
467             // Detected a SSLv3 / TLSv1 packet
468             packetLength = (buffer.getShort(buffer.readerIndex() + 3) & 0xFFFF) + 5;
469         }
470 
471         if (buffer.readableBytes() < packetLength) {
472             return null;
473         }
474 
475         // We advance the buffer's readerIndex before calling unwrap() because
476         // unwrap() can trigger FrameDecoder call decode(), this method, recursively.
477         // The recursive call results in decoding the same packet twice if
478         // the readerIndex is advanced *after* decode().
479         //
480         // Here's an example:
481         // 1) An SSL packet is received from the wire.
482         // 2) SslHandler.decode() deciphers the packet and calls the user code.
483         // 3) The user closes the channel in the same thread.
484         // 4) The same thread triggers a channelDisconnected() event.
485         // 5) FrameDecoder.cleanup() is called, and it calls SslHandler.decode().
486         // 6) SslHandler.decode() will feed the same packet with what was
487         //    deciphered at the step 2 again if the readerIndex was not advanced
488         //    before calling the user code.
489         final int packetOffset = buffer.readerIndex();
490         buffer.skipBytes(packetLength);
491         return unwrap(ctx, channel, buffer, packetOffset, packetLength);
492     }

494     private ChannelFuture wrap(ChannelHandlerContext context, Channel channel)
495             throws SSLException {
496 
497         ChannelFuture future = null;
498         ChannelBuffer msg;
499         ByteBuffer outNetBuf = bufferPool.acquire();
500         boolean success = true;
501         boolean offered = false;
502         boolean needsUnwrap = false;
503         try {
504             loop:
505             for (;;) {
506                 // Acquire a lock to make sure unencrypted data is polled
507                 // in order and their encrypted counterpart is offered in
508                 // order.
509                 synchronized (pendingUnencryptedWrites) {
510                     PendingWrite pendingWrite = pendingUnencryptedWrites.peek();
511                     if (pendingWrite == null) {
512                         break;
513                     }
514 
515                     ByteBuffer outAppBuf = pendingWrite.outAppBuf;
516                     if (outAppBuf == null) {
517                         // A write request with an empty buffer
518                         pendingUnencryptedWrites.remove();
519                         offerEncryptedWriteRequest(
520                                 new DownstreamMessageEvent(
521                                         channel, pendingWrite.future,
522                                         ChannelBuffers.EMPTY_BUFFER,
523                                         channel.getRemoteAddress()));
524                         offered = true;
525                     } else {
526                         SSLEngineResult result = null;
527                         try {
528                             synchronized (handshakeLock) {
529                                 result = engine.wrap(outAppBuf, outNetBuf);
530                             }
531                         } finally {
532                             if (!outAppBuf.hasRemaining()) {
533                                 pendingUnencryptedWrites.remove();
534                             }
535                         }
536 
537                         if (result.bytesProduced() > 0) {
538                             outNetBuf.flip();
539                             msg = ChannelBuffers.buffer(outNetBuf.remaining());
540                             msg.writeBytes(outNetBuf.array(), 0, msg.capacity());
541                             outNetBuf.clear();
542 
543                             if (pendingWrite.outAppBuf.hasRemaining()) {
544                                 // pendingWrite's future shouldn't be notified if
545                                 // only partial data is written.
546                                 future = succeededFuture(channel);
547                             } else {
548                                 future = pendingWrite.future;
549                             }
550 
551                             MessageEvent encryptedWrite = new DownstreamMessageEvent(
552                                     channel, future, msg, channel.getRemoteAddress());
553                             offerEncryptedWriteRequest(encryptedWrite);
554                             offered = true;
555                         } else {
556                             HandshakeStatus handshakeStatus = result.getHandshakeStatus();
557                             switch (handshakeStatus) {
558                             case NEED_WRAP:
559                                 if (outAppBuf.hasRemaining()) {
560                                     break;
561                                 } else {
562                                     break loop;
563                                 }
564                             case NEED_UNWRAP:
565                                 needsUnwrap = true;
566                                 break loop;
567                             case NEED_TASK:
568                                 runDelegatedTasks();
569                                 break;
570                             case FINISHED:
571                             case NOT_HANDSHAKING:
572                                 if (handshakeStatus == HandshakeStatus.FINISHED) {
573                                     setHandshakeSuccess(channel);
574                                 }
575                                 if (result.getStatus() == Status.CLOSED) {
576                                     success = false;
577                                 }
578                                 break loop;
579                             default:
580                                 throw new IllegalStateException(
581                                         "Unknown handshake status: " +
582                                         result.getHandshakeStatus());
583                             }
584                         }
585                     }
586                 }
587             }
588         } catch (SSLException e) {
589             success = false;
590             setHandshakeFailure(channel, e);
591             throw e;
592         } finally {
593             bufferPool.release(outNetBuf);
594 
595             if (offered) {
596                 flushPendingEncryptedWrites(context);
597             }
598 
599             if (!success) {
600                 IllegalStateException cause =
601                     new IllegalStateException("SSLEngine already closed");
602                 // Mark all remaining pending writes as failure if anything
603                 // wrong happened before the write requests are wrapped.
604                 // Please note that we do not call setFailure while a lock is
605                 // acquired, to avoid a potential dead lock.
606                 for (;;) {
607                     PendingWrite pendingWrite;
608                     synchronized (pendingUnencryptedWrites) {
609                         pendingWrite = pendingUnencryptedWrites.poll();
610                         if (pendingWrite == null) {
611                             break;
612                         }
613                     }
614 
615                     pendingWrite.future.setFailure(cause);
616                 }
617             }
618         }
619 
620         if (needsUnwrap) {
621             unwrap(context, channel, ChannelBuffers.EMPTY_BUFFER, 0, 0);
622         }
623 
624         if (future == null) {
625             future = succeededFuture(channel);
626         }
627         return future;
628     }

630     private void offerEncryptedWriteRequest(MessageEvent encryptedWrite) {
631         boolean offered;
632         if (Thread.holdsLock(pendingEncryptedWrites)) {
633             offered = pendingEncryptedWrites.offer(encryptedWrite);
634         } else {
635             synchronized (pendingEncryptedWrites) {
636                 offered = pendingEncryptedWrites.offer(encryptedWrite);
637             }
638         }
639         assert offered;
640     }
641 
642     private void flushPendingEncryptedWrites(ChannelHandlerContext ctx) {
643         // Avoid possible dead lock and data integrity issue
644         // which is caused by cross communication between more than one channel
645         // in the same VM.
646         if (Thread.holdsLock(pendingEncryptedWrites)) {
647             return;
648         }
649 
650         synchronized (pendingEncryptedWrites) {
651             if (pendingEncryptedWrites.isEmpty()) {
652                 return;
653             }
654         }
655 
656         synchronized (pendingEncryptedWrites) {
657             MessageEvent e;
658             while ((e = pendingEncryptedWrites.poll()) != null) {
659                 ctx.sendDownstream(e);
660             }
661         }
662     }

664     private ChannelFuture wrapNonAppData(ChannelHandlerContext ctx, Channel channel) throws SSLException {
665         ChannelFuture future = null;
666         ByteBuffer outNetBuf = bufferPool.acquire();
667 
668         SSLEngineResult result;
669         try {
670             for (;;) {
671                 synchronized (handshakeLock) {
672                     result = engine.wrap(EMPTY_BUFFER, outNetBuf);
673                 }
674 
675                 if (result.bytesProduced() > 0) {
676                     outNetBuf.flip();
677                     ChannelBuffer msg = ChannelBuffers.buffer(outNetBuf.remaining());
678                     msg.writeBytes(outNetBuf.array(), 0, msg.capacity());
679                     outNetBuf.clear();
680 
681                     future = future(channel);
682                     future.addListener(new ChannelFutureListener() {
683                         public void operationComplete(ChannelFuture future)
684                                 throws Exception {
685                             if (future.getCause() instanceof ClosedChannelException) {
686                                 synchronized (ignoreClosedChannelExceptionLock) {
687                                     ignoreClosedChannelException ++;
688                                 }
689                             }
690                         }
691                     });
692 
693                     write(ctx, future, msg);
694                 }
695 
696                 switch (result.getHandshakeStatus()) {
697                 case FINISHED:
698                     setHandshakeSuccess(channel);
699                     runDelegatedTasks();
700                     break;
701                 case NEED_TASK:
702                     runDelegatedTasks();
703                     break;
704                 case NEED_UNWRAP:
705                     if (!Thread.holdsLock(handshakeLock)) {
706                         // unwrap shouldn't be called when this method was
707                         // called by unwrap - unwrap will keep running after
708                         // this method returns.
709                         unwrap(ctx, channel, ChannelBuffers.EMPTY_BUFFER, 0, 0);
710                     }
711                     break;
712                 case NOT_HANDSHAKING:
713                 case NEED_WRAP:
714                     break;
715                 default:
716                     throw new IllegalStateException(
717                             "Unexpected handshake status: " +
718                             result.getHandshakeStatus());
719                 }
720 
721                 if (result.bytesProduced() == 0) {
722                     break;
723                 }
724             }
725         } catch (SSLException e) {
726             setHandshakeFailure(channel, e);
727             throw e;
728         } finally {
729             bufferPool.release(outNetBuf);
730         }
731 
732         if (future == null) {
733             future = succeededFuture(channel);
734         }
735 
736         return future;
737     }

739     private ChannelBuffer unwrap(
740             ChannelHandlerContext ctx, Channel channel, ChannelBuffer buffer, int offset, int length) throws SSLException {
741         ByteBuffer inNetBuf = buffer.toByteBuffer(offset, length);
742         ByteBuffer outAppBuf = bufferPool.acquire();
743 
744         try {
745             boolean needsWrap = false;
746             loop:
747             for (;;) {
748                 SSLEngineResult result;
749                 synchronized (handshakeLock) {
750                     if (initialHandshake && !engine.getUseClientMode() &&
751                         !engine.isInboundDone() && !engine.isOutboundDone()) {
752                         handshake(channel);
753                         initialHandshake = false;
754                     }
755 
756                     try {
757                         result = engine.unwrap(inNetBuf, outAppBuf);
758                     } catch (SSLException e) {
759                         throw e;
760                     }
761 
762                     switch (result.getHandshakeStatus()) {
763                     case NEED_UNWRAP:
764                         if (inNetBuf.hasRemaining() && !engine.isInboundDone()) {
765                             break;
766                         } else {
767                             break loop;
768                         }
769                     case NEED_WRAP:
770                         wrapNonAppData(ctx, channel);
771                         break;
772                     case NEED_TASK:
773                         runDelegatedTasks();
774                         break;
775                     case FINISHED:
776                         setHandshakeSuccess(channel);
777                         needsWrap = true;
778                         break loop;
779                     case NOT_HANDSHAKING:
780                         needsWrap = true;
781                         break loop;
782                     default:
783                         throw new IllegalStateException(
784                                 "Unknown handshake status: " +
785                                 result.getHandshakeStatus());
786                     }
787                 }
788             }
789 
790             if (needsWrap) {
791                 // wrap() acquires pendingUnencryptedWrites first and then
792                 // handshakeLock.  If handshakeLock is already hold by the
793                 // current thread, calling wrap() will lead to a dead lock
794                 // i.e. pendingUnencryptedWrites -> handshakeLock vs.
795                 //      handshakeLock -> pendingUnencryptedLock -> handshakeLock
796                 //
797                 // There is also a same issue between pendingEncryptedWrites
798                 // and pendingUnencryptedWrites.
799                 if (!Thread.holdsLock(handshakeLock) &&
800                     !Thread.holdsLock(pendingEncryptedWrites)) {
801                     wrap(ctx, channel);
802                 }
803             }
804 
805             outAppBuf.flip();
806 
807             if (outAppBuf.hasRemaining()) {
808                 ChannelBuffer frame = ChannelBuffers.buffer(outAppBuf.remaining());
809                 frame.writeBytes(outAppBuf.array(), 0, frame.capacity());
810                 return frame;
811             } else {
812                 return null;
813             }
814         } catch (SSLException e) {
815             setHandshakeFailure(channel, e);
816             throw e;
817         } finally {
818             bufferPool.release(outAppBuf);
819         }
820     }

822     private void runDelegatedTasks() {
823         for (;;) {
824             final Runnable task;
825             synchronized (handshakeLock) {
826                 task = engine.getDelegatedTask();
827             }
828 
829             if (task == null) {
830                 break;
831             }
832 
833             delegatedTaskExecutor.execute(new Runnable() {
834                 public void run() {
835                     synchronized (handshakeLock) {
836                         task.run();
837                     }
838                 }
839             });

871     private void closeOutboundAndChannel(
872             final ChannelHandlerContext context, final ChannelStateEvent e) throws SSLException {
873         if (!e.getChannel().isConnected()) {
874             context.sendDownstream(e);
875             return;
876         }
877 
878         unwrap(context, e.getChannel(), ChannelBuffers.EMPTY_BUFFER, 0, 0);
879         if (!engine.isInboundDone()) {
880             if (sentCloseNotify.compareAndSet(false, true)) {
881                 engine.closeOutbound();
882                 ChannelFuture closeNotifyFuture = wrapNonAppData(context, e.getChannel());
883                 closeNotifyFuture.addListener(
884                         new ClosingChannelFutureListener(context, e));
885                 return;
886             }
887         }
888 
889         context.sendDownstream(e);
