36  /**
37   * Triggers an {@link IdleStateEvent} when a {@link Channel} has not performed
38   * read, write, or both operation for a while.
39   *
40   * <h3>Supported idle states</h3>
41   * <table border="1">
42   * <tr>
43   * <th>Property</th><th>Meaning</th>
44   * </tr>
45   * <tr>
46   * <td>{@code readerIdleTime}</td>
47   * <td>an {@link IdleStateEvent} whose state is {@link IdleState#READER_IDLE}
48   *     will be triggered when no read was performed for the specified period of
49   *     time.  Specify {@code 0} to disable.</td>
50   * </tr>
51   * <tr>
52   * <td>{@code writerIdleTime}</td>
53   * <td>an {@link IdleStateEvent} whose state is {@link IdleState#WRITER_IDLE}
54   *     will be triggered when no write was performed for the specified period of
55   *     time.  Specify {@code 0} to disable.</td>
56   * </tr>
57   * <tr>
58   * <td>{@code allIdleTime}</td>
59   * <td>an {@link IdleStateEvent} whose state is {@link IdleState#ALL_IDLE}
60   *     will be triggered when neither read nor write was performed for the
61   *     specified period of time.  Specify {@code 0} to disable.</td>
62   * </tr>
63   * </table>
64   *
65   * <pre>
66   * // An example that sends a ping message when there is no traffic
67   * // (either inbound or outbound) for 30 seconds.
68   * ChannelPipeline p = ...;
69   * Timer timer = new HashedWheelTimer();
70   * p.addLast("timeout", new IdleStateHandler(timer, 30, 30, 0));
71   * p.addLast("handler", new MyHandler());
72   *
73   * // Handler should handle the IdleStateEvent triggered by IdleStateHandler.
74   * public class MyHandler extends IdleStateAwareChannelHandler {
75   *     public void channelIdle(ChannelHandlerContext ctx, IdleStateEvent e) {
76   *         ctx.getChannel().write(new PingMessage());
77   *     }
78   * }
79   *
80   * // To shut down, call {@link #releaseExternalResources()} or {@link Timer#stop()}.
81   * </pre>
82   *
83   * @author The Netty Project (netty-dev@lists.jboss.org)
84   * @author Trustin Lee (tlee@redhat.com)
85   * @version $Rev: 1715 $, $Date: 2009-09-04 15:57:45 +0900 (ê¸ˆ, 04 9 2009) $
86   *

142     /**
143      * Creates a new instance.
144      *
145      * @param timer
146      *        the {@link Timer} that is used to trigger the scheduled event.
147      *        The recommended {@link Timer} implementation is {@link HashedWheelTimer}.
148      * @param readerIdleTime
149      *        an {@link IdleStateEvent} whose state is {@link IdleState#READER_IDLE}
150      *        will be triggered when no read was performed for the specified
151      *        period of time.  Specify {@code 0} to disable.
152      * @param writerIdleTime
153      *        an {@link IdleStateEvent} whose state is {@link IdleState#WRITER_IDLE}
154      *        will be triggered when no write was performed for the specified
155      *        period of time.  Specify {@code 0} to disable.
156      * @param allIdleTime
157      *        an {@link IdleStateEvent} whose state is {@link IdleState#ALL_IDLE}
158      *        will be triggered when neither read nor write was performed for
159      *        the specified period of time.  Specify {@code 0} to disable.
160      * @param unit
161      *        the {@link TimeUnit} of {@code readerIdleTime},
162      *        {@code writeIdleTime}, and {@code allIdleTime}
163      */
164     public IdleStateHandler(
165             Timer timer,
166             long readerIdleTime, long writerIdleTime, long allIdleTime,
167             TimeUnit unit) {
168 
169         if (timer == null) {
170             throw new NullPointerException("timer");
171         }
172         if (unit == null) {
173             throw new NullPointerException("unit");
174         }
175 
176         this.timer = timer;
177         if (readerIdleTime <= 0) {
178             readerIdleTimeMillis = 0;
179         } else {
180             readerIdleTimeMillis = Math.max(unit.toMillis(readerIdleTime), 1);
181         }
182         if (writerIdleTime <= 0) {
183             writerIdleTimeMillis = 0;
184         } else {
185             writerIdleTimeMillis = Math.max(unit.toMillis(writerIdleTime), 1);
186         }
187         if (allIdleTime <= 0) {
188             allIdleTimeMillis = 0;
189         } else {
190             allIdleTimeMillis = Math.max(unit.toMillis(allIdleTime), 1);
191         }
192     }

249     private void initialize(ChannelHandlerContext ctx) {
250         lastReadTime = lastWriteTime = System.currentTimeMillis();
251         if (readerIdleTimeMillis > 0) {
252             readerIdleTimeout = timer.newTimeout(
253                     new ReaderIdleTimeoutTask(ctx),
254                     readerIdleTimeMillis, TimeUnit.MILLISECONDS);
255         }
256         if (writerIdleTimeMillis > 0) {
257             writerIdleTimeout = timer.newTimeout(
258                     new WriterIdleTimeoutTask(ctx),
259                     writerIdleTimeMillis, TimeUnit.MILLISECONDS);
260         }
261         if (allIdleTimeMillis > 0) {
262             allIdleTimeout = timer.newTimeout(
263                     new AllIdleTimeoutTask(ctx),
264                     allIdleTimeMillis, TimeUnit.MILLISECONDS);
265         }
266     }
267 
268     private void destroy() {
269         if (readerIdleTimeout != null) {
270             readerIdleTimeout.cancel();
271             readerIdleTimeout = null;
272         }
273         if (writerIdleTimeout != null) {
274             writerIdleTimeout.cancel();
275             writerIdleTimeout = null;
276         }
277         if (allIdleTimeout != null) {
278             allIdleTimeout.cancel();
279             allIdleTimeout = null;
280         }
281     }

288     private final class ReaderIdleTimeoutTask implements TimerTask {
289 
290         private final ChannelHandlerContext ctx;
291 
292         ReaderIdleTimeoutTask(ChannelHandlerContext ctx) {
293             this.ctx = ctx;
294         }
295 
296         public void run(Timeout timeout) throws Exception {
297             if (timeout.isCancelled() || !ctx.getChannel().isOpen()) {
298                 return;
299             }
300 
301             long currentTime = System.currentTimeMillis();
302             long lastReadTime = IdleStateHandler.this.lastReadTime;
303             long nextDelay = readerIdleTimeMillis - (currentTime - lastReadTime);
304             if (nextDelay <= 0) {
305                 // Reader is idle - set a new timeout and notify the callback.
306                 readerIdleTimeout =
307                     timer.newTimeout(this, readerIdleTimeMillis, TimeUnit.MILLISECONDS);
308                 try {
309                     channelIdle(ctx, IdleState.READER_IDLE, lastReadTime);
310                 } catch (Throwable t) {
311                     fireExceptionCaught(ctx, t);
312                 }
313             } else {
314                 // Read occurred before the timeout - set a new timeout with shorter delay.
315                 readerIdleTimeout =
316                     timer.newTimeout(this, nextDelay, TimeUnit.MILLISECONDS);
317             }
318         }
319 
320     }

323 
324         private final ChannelHandlerContext ctx;
325 
326         WriterIdleTimeoutTask(ChannelHandlerContext ctx) {
327             this.ctx = ctx;
328         }
329 
330         public void run(Timeout timeout) throws Exception {
331             if (timeout.isCancelled() || !ctx.getChannel().isOpen()) {
332                 return;
333             }
334 
335             long currentTime = System.currentTimeMillis();
336             long lastWriteTime = IdleStateHandler.this.lastWriteTime;
337             long nextDelay = writerIdleTimeMillis - (currentTime - lastWriteTime);
338             if (nextDelay <= 0) {
339                 // Writer is idle - set a new timeout and notify the callback.
340                 writerIdleTimeout =
341                     timer.newTimeout(this, writerIdleTimeMillis, TimeUnit.MILLISECONDS);
342                 try {
343                     channelIdle(ctx, IdleState.WRITER_IDLE, lastWriteTime);
344                 } catch (Throwable t) {
345                     fireExceptionCaught(ctx, t);
346                 }
347             } else {
348                 // Write occurred before the timeout - set a new timeout with shorter delay.
349                 writerIdleTimeout =
350                     timer.newTimeout(this, nextDelay, TimeUnit.MILLISECONDS);
351             }
352         }
353     }

355     private final class AllIdleTimeoutTask implements TimerTask {
356 
357         private final ChannelHandlerContext ctx;
358 
359         AllIdleTimeoutTask(ChannelHandlerContext ctx) {
360             this.ctx = ctx;
361         }
362 
363         public void run(Timeout timeout) throws Exception {
364             if (timeout.isCancelled() || !ctx.getChannel().isOpen()) {
365                 return;
366             }
367 
368             long currentTime = System.currentTimeMillis();
369             long lastIoTime = Math.max(lastReadTime, lastWriteTime);
370             long nextDelay = allIdleTimeMillis - (currentTime - lastIoTime);
371             if (nextDelay <= 0) {
372                 // Both reader and writer are idle - set a new timeout and
373                 // notify the callback.
374                 allIdleTimeout =
375                     timer.newTimeout(this, allIdleTimeMillis, TimeUnit.MILLISECONDS);
376                 try {
377                     channelIdle(ctx, IdleState.ALL_IDLE, lastIoTime);
378                 } catch (Throwable t) {
379                     fireExceptionCaught(ctx, t);
380                 }
381             } else {
382                 // Either read or write occurred before the timeout - set a new
383                 // timeout with shorter delay.
384                 allIdleTimeout =
385                     timer.newTimeout(this, nextDelay, TimeUnit.MILLISECONDS);
386             }
387         }
