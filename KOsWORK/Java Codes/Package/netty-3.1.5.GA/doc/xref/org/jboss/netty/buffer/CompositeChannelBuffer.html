49      public CompositeChannelBuffer(ChannelBuffer... buffers) {
50          if (buffers.length == 0) {
51              throw new IllegalArgumentException("buffers should not be empty.");
52          }
53  
54          ByteOrder expectedEndianness = null;
55          for (ChannelBuffer buffer : buffers) {
56              if (buffer.capacity() != 0) {
57                  expectedEndianness = buffer.order();
58              }
59          }
60  
61          if (expectedEndianness == null) {
62              throw new IllegalArgumentException("buffers have only empty buffers.");
63          }
64  
65          order = expectedEndianness;
66          slices = new ChannelBuffer[buffers.length];
67          for (int i = 0; i < buffers.length; i ++) {
68              if (buffers[i].capacity() != 0 && buffers[i].order() != expectedEndianness) {
69                  throw new IllegalArgumentException(
70                          "All buffers must have the same endianness.");
71              }
72              slices[i] = buffers[i].slice();
73          }
74          indices = new int[buffers.length + 1];
75          for (int i = 1; i <= buffers.length; i ++) {
76              indices[i] = indices[i - 1] + slices[i - 1].capacity();
77          }
78          writerIndex(capacity());
79      }

81      private CompositeChannelBuffer(CompositeChannelBuffer buffer) {
82          order = buffer.order;
83          slices = buffer.slices.clone();
84          indices = buffer.indices.clone();
85          setIndex(buffer.readerIndex(), buffer.writerIndex());
86      }

105     public short getShort(int index) {
106         int sliceId = sliceId(index);
107         if (index + 2 <= indices[sliceId + 1]) {
108             return slices[sliceId].getShort(index - indices[sliceId]);
109         } else if (order() == ByteOrder.BIG_ENDIAN) {
110             return (short) ((getByte(index) & 0xff) << 8 | getByte(index + 1) & 0xff);
111         } else {
112             return (short) (getByte(index) & 0xff | (getByte(index + 1) & 0xff) << 8);
113         }
114     }

149     public void getBytes(int index, byte[] dst, int dstIndex, int length) {
150         int sliceId = sliceId(index);
151         if (index > capacity() - length || dstIndex > dst.length - length) {
152             throw new IndexOutOfBoundsException();
153         }
154 
155         int i = sliceId;
156         while (length > 0) {
157             ChannelBuffer s = slices[i];
158             int adjustment = indices[i];
159             int localLength = Math.min(length, s.capacity() - (index - adjustment));
160             s.getBytes(index - adjustment, dst, dstIndex, localLength);
161             index += localLength;
162             dstIndex += localLength;
163             length -= localLength;
164             i ++;
165         }
166     }

359     public int setBytes(int index, InputStream in, int length)
360             throws IOException {
361         int sliceId = sliceId(index);
362         if (index > capacity() - length) {
363             throw new IndexOutOfBoundsException();
364         }
365 
366         int i = sliceId;
367         int readBytes = 0;
368 
369         do {
370             ChannelBuffer s = slices[i];
371             int adjustment = indices[i];
372             int localLength = Math.min(length, s.capacity() - (index - adjustment));
373             int localReadBytes = s.setBytes(index - adjustment, in, localLength);
374             if (localReadBytes < 0) {
375                 if (readBytes == 0) {
376                     return -1;
377                 } else {
378                     break;
379                 }
380             }
381 
382             if (localReadBytes == localLength) {
383                 index += localLength;
384                 length -= localLength;
385                 readBytes += localLength;
386                 i ++;
387             } else {
388                 index += localReadBytes;
389                 length -= localReadBytes;
390                 readBytes += localReadBytes;
391             }
392         } while (length > 0);
393 
394         return readBytes;
395     }

462     public ChannelBuffer slice(int index, int length) {
463         if (index == 0) {
464             if (length == 0) {
465                 return ChannelBuffers.EMPTY_BUFFER;
466             } else {
467                 return new TruncatedChannelBuffer(this, length);
468             }
469         } else if (index < 0 || index > capacity() - length) {
470             throw new IndexOutOfBoundsException();
471         } else if (length == 0) {
472             return ChannelBuffers.EMPTY_BUFFER;
473         } else {
474             return new SlicedChannelBuffer(this, index, length);
475         }

478     public ByteBuffer toByteBuffer(int index, int length) {
479         if (slices.length == 1) {
480             return slices[0].toByteBuffer(index, length);
481         }
482 
483         ByteBuffer[] buffers = toByteBuffers(index, length);
484         ByteBuffer merged = ByteBuffer.allocate(length).order(order());
485         for (ByteBuffer b: buffers) {
486             merged.put(b);
487         }
488         merged.flip();
489         return merged;
490     }

492     @Override
493     public ByteBuffer[] toByteBuffers(int index, int length) {
494         int sliceId = sliceId(index);
495         if (index + length > capacity()) {
496             throw new IndexOutOfBoundsException();
497         }
498 
499         List<ByteBuffer> buffers = new ArrayList<ByteBuffer>(slices.length);
500 
501         int i = sliceId;
502         while (length > 0) {
503             ChannelBuffer s = slices[i];
504             int adjustment = indices[i];
505             int localLength = Math.min(length, s.capacity() - (index - adjustment));
506             buffers.add(s.toByteBuffer(index - adjustment, localLength));
507             index += localLength;
508             length -= localLength;
509             i ++;
510         }
511 
512         return buffers.toArray(new ByteBuffer[buffers.size()]);
513     }

515     public String toString(int index, int length, String charsetName) {
516         int sliceId = sliceId(index);
517         if (index + length <= indices[sliceId + 1]) {
518             return slices[sliceId].toString(
519                     index - indices[sliceId], length, charsetName);
520         }
521 
522         byte[] data = new byte[length];
523         int dataIndex = 0;
524         int i = sliceId;
525 
526         while (length > 0) {
527             ChannelBuffer s = slices[i];
528             int adjustment = indices[i];
529             int localLength = Math.min(length, s.capacity() - (index - adjustment));
530             s.getBytes(index - adjustment, data, dataIndex, localLength);

534             i ++;
535         }

544     private int sliceId(int index) {
545         int lastSliceId = this.lastSliceId;
546         if (index >= indices[lastSliceId]) {
547             if (index < indices[lastSliceId + 1]) {
548                 return lastSliceId;
549             }
550 
551             // Search right
552             for (int i = lastSliceId + 1; i < slices.length; i ++) {
553                 if (index < indices[i + 1]) {
554                     this.lastSliceId = i;
555                     return i;
556                 }
557             }
558         } else {
559             // Search left
560             for (int i = lastSliceId - 1; i >= 0; i --) {
561                 if (index >= indices[i]) {
562                     this.lastSliceId = i;
563                     return i;
564                 }
565             }
566         }
567 
568         throw new IndexOutOfBoundsException();
569     }
